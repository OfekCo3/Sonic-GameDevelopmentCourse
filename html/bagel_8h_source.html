52 template <class T, int N>
53 class DynamicBag : NoCopy
54 {
55 public:
56 void push(const T& t) {
57 if (_size == _capacity) {
58 _capacity *= 2;
59 _arr = static_cast<T*>(
60 realloc(_arr, sizeof(T)*_capacity));
61 }
62 _arr[_size] = t;
63 ++_size;
64 }
65 void ensure(size_type s) {
66 if (_capacity < s) {
67 _capacity = std::max(s, _capacity*2);
68 _arr = static_cast<T*>(
69 realloc(_arr, sizeof(T)*_capacity));
70 }
71 }
72 T pop() { return _arr[--_size]; }
73 T& operator[](index_type i) { return _arr[i]; }
74 const T& operator[](index_type i) const { return _arr[i]; }
75 void clear() { _size = 0; }
76
77 size_type size() const { return _size; }
78 size_type capacity() const { return _capacity; }
79
80 ~DynamicBag() { free(_arr); }
81 private:
82 T* _arr = static_cast<T*>(malloc(sizeof(T) * N));
83 size_type _size = 0;
84 size_type _capacity = N;
85 };

119 class PackedStorage final : NoInstance
120 {
121 public:
122 static void add(ent_type e, const T& t) {
123 _entToComp.ensure(e.id);
124 _entToComp[e.id] = _comps.size();
125 _comps.push(t);
126 _compToEnt.push(e);
127 }
128 static void del(ent_type e) {
129 index_type ent_comp_idx = _entToComp[e.id];
130 ent_type last_ent = _compToEnt.pop();
131
132 _comps[ent_comp_idx] = _comps.pop();
133 _compToEnt[ent_comp_idx] = last_ent;
134 _entToComp[last_ent.id] = ent_comp_idx;
135 }
136 static T& get(ent_type e) {
137 return _comps[_entToComp[e.id]];
138 }
139 static int size() { return _comps.size(); }
140 static T& get(index_type idx) {
141 return _comps[idx];
142 }
143 static ent_type entity(index_type idx) {
144 return _compToEnt[idx];
145 }
146 private:
147 static inline Bag<T,Params.InitialPackedSize> _comps;

181 class MultiMask final
182 {
183 public:
184 using bit_type = struct {
185 const index_type index;
186 const mask_type mask;

188 static constexpr bit_type bit(index_type idx) {
189 return {idx/BitsetWidth, static_cast<mask_type>(1<<(idx%BitsetWidth))};
190 }

192 void set(const bit_type& b) { _masks[b.index] |= b.mask; }

194 void clear(const bit_type& b) { _masks[b.index] &= ~b.mask; }

197 bool test(const bit_type& b) const { return _masks[b.index] & b.mask; }
198 bool test(const MultiMask& m) const {
199 for (index_type i = 0; i < Size; ++i)
200 if ((_masks[i] & m._masks[i]) != m._masks[i])
201 return false;
202 return true;
203 }
